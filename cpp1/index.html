數學: 先乘再加減

            
            # ★ 先乘再加減
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print(f"{a}*{b}+{c} = {a*b+c}")
            print(f"{a}*{b}-{c} = {a*b-c}")

數學: 先加減再乘

            
            # ★ 先加減再乘
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print(f"({a}+{b})*{c} = {(a+b)*c}")
            print(f"({a}-{b})*{c} = {(a-b)*c}")
