迴圈．數列．圖形

for迴圈: 輸出 1~5

            
            # ★ 輸出 1~5
            for i in range(1, 5+1):
                print(i)

for迴圈: 乘法表

            
            # ★ 乘法表
            n = int(input("n: "))
            for i in range(1, 10+1):
                print(f'{n} x {i} = {n*i}')

for迴圈: 等差數列 (1/2)

            
            # ★ 等差數列
            for i in range(1, 20+1, 2):
                print(i)

for迴圈: 等差數列 (2/2)

            
            # ★ 等差數列
            a = int(input("a: "))
            b = int(input("b: "))
            d = int(input("d: "))
            for i in range(a, b+1, d):
                print(i)

for迴圈: 1~5 和 5~1

            
            # ★ 1~5 和 5~1
            for i in range(1, 5+1):
                print(i, 5-i+1)

for迴圈: 1~n 和 n~1

            
            # ★ 1~n 和 n~1
            n = int(input("n: "))
            for i in range(1, n+1):
                print(i, n-i+1)

字串: 字串加法

            
            # ★ 字串加法
            # 兩個字串，合併或串接
            # 運算式: 字串+字串
            str1 = input("字串1: ")
            str2 = input("字串2: ")
            print(str1 + str2)

字串: 字串乘法

            
            # ★ 字串乘法
            # 一個字串複製多個，合併或串接
            # 運算式: 字串*整數
            astr = input("字串: ")
            n = int(input("複製幾個: "))
            print(astr*n)

圖形: 正方形

            
            # ★ 正方形
            length = int(input("長: "))
            for i in range(length):
                print("O"*length)

圖形: 長方形

            
            # ★ 長方形
            length = int(input("長: "))
            width = int(input("寬: "))
            for i in range(length):
                print("O"*width)
