輸出、輸入、計算、變數

輸出: Hello1(哈囉1)

            
            print("Hello, Bug!")
            print("Hello, Ashley!")

輸出: fruit(水果)

            
            print("apple")
            print("lemon")
            print("cherry")
            print("jujube")

輸出: animal(動物)

            
            print("dog", "cow", "wolf")
            print("cat", "deer", "zebra")

輸入: Hello2(哈囉2)

            
            name1 = input("Name 1: ")
            print(f"Hello, {name1}!")
            name2 = input("Name 2: ")
            print(f"Hello, {name2}!")

輸入: 造句1

            
            # 姐姐的頭髮越來越長
            who = input("誰? ")
            what = input("什麼? ")
            how = input("怎樣? ")
            print(f"{who}的{what}越來越{how}")

數學: 加法和減法1

            
            a = int(input("a: "))
            b = int(input("b: "))
            print(a+b)
            print(a-b)

數學: 加法和減法2

            
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print(a+b-c)
            print(a-b+c)

輸入: 造句2

            
            # 小鳥在樹枝上唱歌
            who = input("誰? ")
            where = input("哪裡? ")
            what = input("做什麼? ")
            print(f"{who}在{where}{what}")

輸入: 造句3

            
            # 姐姐跟媽媽去逛街
            who = input("誰? ")
            with_who = input("跟誰? ")
            what = input("做什麼? ")
            print(f"{who}跟{with_who}{what}")

數學: 加減法等式1

            
            a = int(input("a: "))
            b = int(input("b: "))
            print(f"{a}+{b}={a+b}")
            print(f"{a}-{b}={a-b}")

數學: 加減法等式2

            
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print(f"{a}+{b}-{c}={a+b-c}")
            print(f"{a}-{b}+{c}={a-b+c}")
