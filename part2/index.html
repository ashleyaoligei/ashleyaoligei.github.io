選擇: 剪刀石頭布

            
            # ★ 剪刀石頭布
            import random
            alist = ['rock', 'paper', 'scissors']
            player = input("要出什麼? ")
            computer = random.choice(alist)
            if player == computer:
                print("平手！")
            elif player == "rock" and computer == "scissors":
                print("贏了！")
            elif player == "paper" and computer == "rock":
                print("贏了！")
            elif player == "scissors" and computer == "paper":
                print("贏了！")
            else:
                print("輸給電腦了！")
