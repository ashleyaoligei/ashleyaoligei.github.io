Part 2: 比較．邏輯．條件．選擇

比較: 是否相等

            
            # ★ 兩個數字一樣嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a == b)

比較: 是否不相等

            
            # ★ 兩個數字不一樣嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a != b)

比較: 是否大於

            
            # ★ a 有比 b 大嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a > b)

比較: 是否小於

            
            # ★ a 有比 b 小嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a < b)

比較: 是否大於等於

            
            # ★ a 有大於等於 b 嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a >= b)

比較: 是否小於等於

            
            # ★ a 有小於等於 b 嗎？
            a = int(input("a: "))
            b = int(input("b: "))
            print(a <= b)

邏輯: and運算1(而且1)

            
            # ★ 哪個數字最大
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print("a", a > b and a > c)
            print("b", b > a and b > c)
            print("c", c > a and c > b)

邏輯: and運算2(而且2)

            
            # ★ 哪個數字最小
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print("a", a < b and a < c)
            print("b", b < a and b < c)
            print("c", c < a and c < b)

邏輯: or運算1(或者1)

            
            # ★ a 或 b 有比 x 大嗎？
            x = int(input("x: "))
            a = int(input("a: "))
            b = int(input("b: "))
            print(a > x or b > x)

邏輯: or運算2(或者2)

            
            # ★ a 或 b 有比 x 小嗎？
            x = int(input("x: "))
            a = int(input("a: "))
            b = int(input("b: "))
            print(a < x or b < x)

邏輯: 撒花1

            
            # ★ 撒花: 男生 boy 女生 girl
            print("老師請班長找兩個同學撒花")
            print("兩個都要女生")
            who1 = input("第 1 個是? ")
            who2 = input("第 2 個是? ")
            print(who1 == "girl" and who2 == "girl")

邏輯: 撒花2

            
            # ★ 撒花: 男生 boy 女生 girl
            print("老師請班長找兩個同學撒花")
            print("至少一個女生")
            who1 = input("第 1 個是? ")
            who2 = input("第 2 個是? ")
            print(who1 == "girl" or who2 == "girl")

邏輯: 撒花3

            
            # ★ 撒花: 男生 boy 女生 girl
            print("老師請班長找兩個同學撒花")
            print("一定要有男生女生")
            who1 = input("第 1 個是? ")
            who2 = input("第 2 個是? ")
            print(who1 != who2)

選擇: 最大數

            
            # ★ 哪個數字最大
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            if a > b and a > c:
                print(a, "最大")
            elif b > a and b > c:
                print(b, "最大")
            elif c > a and c > b:
                print(c, "最大")

選擇: 最小數

            
            # ★ 哪個數字最小
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            if a < b and a < c:
                print(a, "最小")
            elif b < a and b < c:
                print(b, "最小")
            elif c < a and c < b:
                print(c, "最小")

選擇: 商數與餘數

            
            # ★ 商數與餘數
            a = int(input("a: "))
            b = int(input("b: "))
            print(f"{a} 除以 {b}")
            print(f"商數是 {a//b}")
            print(f"餘數是 {a%b}")

選擇: 奇數與偶數

            
            # ★ 奇數與偶數
            n = int(input("n: "))
            if n % 2 == 0:
                print(f"{n} 是偶數")
            elif n % 2 == 1:
                print(f"{n} 是奇數")

選擇: 倍數

            
            # ★ 倍數
            n = int(input("n: "))
            if n % 3 == 0:
                print(f"{n} 是 3 的倍數")
            elif n % 5 == 0:
                print(f"{n} 是 5 的倍數")
            elif n % 7 == 0:
                print(f"{n} 是 7 的倍數")
