輸出: Hello, Ashley!

輸出: animal(動物)

            
            -- ★ animal(動物)
            -- print(): 輸出函式
            -- string.format("格式字串", 參數...)
            -- %s:字串, %d:整數, %f浮點數
            print(string.format("%s %s %s", "dog", "cow", "wolf"))
            print(string.format("%s %s %s", "cat", "deer", "zebra"))

輸入輸出: Hello, Ashley!

            
            -- ★ Hello(哈囉)
            -- local 宣告區域變數
            -- io.write(): 提示輸入函式
            -- io.read(): 接收輸入函式
            -- "字串"..字串變數.."字串"
            local name1, name2
            io.write("Name1: ")
            name1 = io.read()
            print("Hello, "..name1.."!")
            io.write("Name2: ")
            name2 = io.read()
            print("Hello, "..name2.."!")

輸入輸出: 造句1 (姐姐的頭髮越來越長)

            
            -- ★ 造句: 姐姐的頭髮越來越長
            -- input(): 輸入函式
            -- 變數: 裝字串或整數
            -- f-string: 可將變數放到字串裡面
            -- () 圓括 {} 花括
            who = input("誰? ")
            what = input("什麼? ")
            how = input("怎樣? ")
            print(f"{who}的{what}越來越{how}")

數學: 加減法等式 (2/2)

            b = int(input("b: "))

            print(f"{a}+{b}-{c} = {a+b-c}")

數學: 先乘再加減

            b = int(input("b: "))

            print(f"{a}*{b}+{c} = {a*b+c}")
            print(f"{a}*{b}-{c} = {a*b-c}")

數學: 先加減再乘

            
            -- ★ 先加減再乘
            a = int(input("a: "))
            b = int(input("b: "))
            c = int(input("c: "))
            print(f"({a}+{b})*{c} = {(a+b)*c}")
            print(f"({a}-{b})*{c} = {(a-b)*c}")
